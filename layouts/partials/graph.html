{{- $currentPage := . -}} {{- $currentPath := .RelPermalink -}} {{- /* Generate graph data with tags as nodes */ -}} {{- $nodes := slice -}} {{- $links := slice -}} {{- $titleToPage := dict -}} {{- $addedTags := slice -}} {{- /* Build page lookup */ -}} {{- range .Site.RegularPages -}} {{- $titleToPage = merge $titleToPage (dict .Title .) -}} {{- end -}} {{- /* Add page nodes */ -}} {{- range .Site.RegularPages -}} {{- $nodes = $nodes | append (dict "id" .RelPermalink "title" .Title "type" "page") -}} {{- end -}} {{- /* Add tag nodes and links */ -}} {{- range .Site.RegularPages -}} {{- $source := .RelPermalink -}} {{- $content := .RawContent -}} {{- $currentTags := .Params.tags -}} {{- /* Wiki links */ -}} {{- $pattern := `\[\[([^\]|]+)(?:\|[^\]]+)?\]\]` -}} {{- range $match := findRE $pattern $content -}} {{- $linkTitle := replaceRE $pattern "$1" $match -}} {{- $targetPage := index $titleToPage $linkTitle -}} {{- if $targetPage -}} {{- $links = $links | append (dict "source" $source "target" $targetPage.RelPermalink "type" "direct") -}} {{- end -}} {{- end -}} {{- /* Tag links - connect pages to tag nodes */ -}} {{- if $currentTags -}} {{- range $tag := $currentTags -}} {{- $tagId := printf "tag:%s" $tag -}} {{- if not (in $addedTags $tagId) -}} {{- $nodes = $nodes | append (dict "id" $tagId "title" $tag "type" "tag") -}} {{- $addedTags = $addedTags | append $tagId -}} {{- end -}} {{- $links = $links | append (dict "source" $source "target" $tagId "type" "tag") -}} {{- end -}} {{- end -}} {{- end -}}
